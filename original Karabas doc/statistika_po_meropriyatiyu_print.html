Статистика по мероприятию

 
Команда возвращает текущее состояние свободных и проданных мест по мероприятию
 
GetEventStatusCommand.cmd
Параметры
event
ИД мероприятия
Обязательный параметр
format
По умолчанию JSON, или XML если передать format=xml
Формат выдачи данных
__auth
Ключ авторизации
Обязательный параметр
Ответ
prices
 
Массив статистики
Ниже следуют аттрибуты объектов массива:
price
Ценовая группа
 
free
Кол-во доступных к продаже мест
 
color
Цвет для отрисовки легенды
 
Пример кода:
 
function loadEventStats() {
    var src = "http://kassir.kartina.tv/GetEventStatusCommand.cmd"
    $.post(src, { __auth: _auth, event: _event }, function (data) {
            $("#eventStats").html("");
            var table = $("<table></table>").appendTo("#eventStats");
 
            for (var i = 0; i < data.prices.length; i++) {
                var price = data.prices[i];
                var tr = $("<tr></tr>").appendTo(table);
                $("<td></td>").html("&nbsp;").css("background-color", price.color).appendTo(tr);
                $("<td></td>").html(price.free).appendTo(tr);
                $("<td></td>").html(price.price).appendTo(tr);
            }
 
        }, "jsonp");
    }
}
